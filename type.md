# tuple,list,set,dict

```python
#!/usr/bin/python
# encoding=utf-8

t=tuple('aaa')
t1=('aaa',30)
print('--tuple--')
print(t)
print(t1)
print()

l=list('aaa')
l1=['aaa',30]
print('--list--')
print(l)
print(l1)
print()

s=set('aaa')
s1={'aaa',30}
print('--set--')
print(s)
print(s1)
print()

d = dict(name='aaa', age=30)
d1 = {'name':'aaa', 'age':30}
print('--dict--')
print(d)
print(d1)
print()
```
